USING (VALUES
    (N'ser', N'Submitter'),
    (N'edr', N'Admin')
) AS s (FormArea, Role)
ON t.FormArea = s.FormArea AND t.Role = s.Role
WHEN NOT MATCHED THEN INSERT (FormArea, Role) VALUES (s.FormArea, s.Role);
